		Root = {

					is_hostile = from

				}
			}
			solar_system = {
				any_fleet_in_system = {
					exists = leader
					leader = {
						owner = {
							is_same_value = from
						}
						has_trait = leader_trait_fly
					}
				}
			}
		}
	}
}

					}
