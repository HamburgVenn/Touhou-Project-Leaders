touhou_trace_select_effect = {
	potential = {
		always = yes
	}
	allow = {
		root = {
			exists = owner
			owner = {
				is_same_value = from
			}
			exists = leader
			leader = {
				has_touhou_leader_trait = yes
			}
		}
	}

	effect = {
		custom_tooltip = touhou_trace_select_desc
		hidden_effect = {
			from = {
				set_variable = {
					which = seleted_char_id
					value = root.value:touhou_leader_id
				}
			}
			if = {
				limit = {
					check_variable_arithmetic = {
						which = value:leader_num_spell_cards
						value > 0
					}
				}
				from = {
					set_variable = {
						which = seleted_card_id
						value = 1
					}
					set_variable = {
						which = seleted_card_nums
						value = root.value:leader_num_spell_cards
					}
				}
			}
			else = {
				from = {
					set_variable = {
						which = seleted_card_id
						value = 0
					}
					set_variable = {
						which = seleted_card_nums
						value = 0
					}
				}
			}
		}
	}
}

touhou_reimu_spell_card_effect = {
	potential = {
		from = {
			check_variable = {
				which = seleted_char_id
				value = 1
			}
			any_owned_leader = {
				has_trait = leader_trait_fly
			}
		}
	}
	allow = {
		hidden_trigger = {
			OR = {
				AND = {
					From = {
						check_variable = {
							which = seleted_card_id
							value = 1
						}
						OR = {
							AND = {
								resource_stockpile_compare = {
									resource = thl_bomb
									value >= 1
								}
								check_variable = {
									which = ex_charge
									value = 0
								}
							}
							resource_stockpile_compare = {
								resource = thl_bomb
								value >= 2
							}
						}
					}
					reimu_card_1_trigger = yes
				}
				AND = {
					From = {
						check_variable = {
							which = seleted_card_id
							value = 2
						}
					}
					reimu_card_2_trigger = yes
					resource_stockpile_compare = {
						resource = thl_bomb
						value >= 2
					}
				}
				AND = {
					From = {
						check_variable = {
							which = seleted_card_id
							value = 3
						}
					}
					reimu_card_3_trigger = yes
					resource_stockpile_compare = {
						resource = thl_bomb
						value >= 1
					}
				}
				AND = {
					From = {
						check_variable = {
							which = seleted_card_id
							value = 4
						}
					}
					reimu_card_4_trigger = yes
					resource_stockpile_compare = {
						resource = thl_bomb
						value >= 4
					}
				}
				AND = {

							value = 5
						}

						resource = thl_bomb
						value >= 1
					}
				}

						check_variable = {

							value = 6
						}
					}
					reimu_card_6_trigger = yes
					resource_stockpile_compare = {
						resource = thl_bomb
						value >= 5
					}
				}
			}
		}
	}

	effect = {
		hidden_effect = {
			if = {

					From = {
						check_variable = {
							which = seleted_card_id
							value = 1
						}
						check_variable = {
							which = ex_charge
							value = 0
						}
					}
				}
				reimu_spell_card_1_effect = yes
			}
			else_if = {
				limit = {
					From = {
						check_variable = {
							which = seleted_card_id
							value = 1
						}
						check_variable = {
							which = ex_charge
							value = 1
						}
					}
				}
				reimu_spell_card_1_ex_effect = yes
			}
			else_if = {
				limit = {
					From = {
						check_variable = {
							which = seleted_card_id
							value = 2
						}
					}
				}
				reimu_spell_card_2_effect = yes
			}
			else_if = {
				limit = {
					From = {
						check_variable = {
							which = seleted_card_id
							value = 3
						}
					}
				}
				reimu_spell_card_3_effect = yes
			}
